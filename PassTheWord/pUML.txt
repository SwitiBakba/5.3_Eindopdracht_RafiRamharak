package "PassTheWord" {
    class Program <<Client>> {
        - Main(args : string[]) : void
    }

    class PasswordService <<Facade>> {
        - _optionsValidator : PasswordOptionsValidator
        - _generationFactory : PasswordGenerationFactory
        - _replacementService : CharacterReplacementService
        - _externalValidationService : ExternalPasswordValidationService
        - _logger : ILogger

        + PasswordService(logger : ILogger)
        + AddExternalVerifier(verifier : IExternalPasswordVerifier) : void
        + GeneratePassword(options : PasswordOptions) : string
    }

    class PasswordOptions {
        + MinimumLength : int
        + MaximumLength : int
        + UseUppercase : bool
        + UseLowercase : bool
        + UseDigits : bool
        + UseSymbols : bool
        + RequireUppercase : bool
        + RequireDigit : bool
        + RequireSymbol : bool
        + ExcludeSimilarCharacters : bool
        + Alphabets : List<IAlphabet>
        + Replacements : Dictionary<char, char>
        + DictionaryWords : List<string>?
        + UseDictionary : bool
    }
}

package "Logging" {
    enum LogLevel {
        None
        Warnings
        All
    }

    interface ILogger {
        + Level : LogLevel
        + Warning(message : string) : void
        + Info(message : string) : void
    }

    class ConsoleLogger {
        + Level : LogLevel
        + ConsoleLogger(logLevel : LogLevel)
        + Warning(message : string) : void
        + Info(message : string) : void
    }
}

package "Alphabets" {
    interface IAlphabet {
        + Name : string
        + UppercaseCharacters : string
        + LowercaseCharacters : string
    }

    class LatinAlphabet {
        + Name : string
        + UppercaseCharacters : string
        + LowercaseCharacters : string
    }

    class CyrillicAlphabet {
        + Name : string
        + UppercaseCharacters : string
        + LowercaseCharacters : string
    }
}

package "Replacements" {
    class CharacterReplacementService {
        + Apply(password : string, replacements : IReadOnlyDictionary<char, char>) : string
    }
}

package "Generation - Strategy Pattern" {
    interface IPasswordGenerationStrategy <<Strategy>> {
        + Generate(options : PasswordOptions) : string
    }

    class PasswordGenerationFactory <<Factory>> {
        + Create(options : PasswordOptions) : IPasswordGenerationStrategy
    }

    class RandomPasswordStrategy <<ConcreteStrategy>> {
        - Digits : string
        - DigitsWithoutSimilar : string
        - Symbols : string
        - SimilarCharacters : string

        + Generate(options : PasswordOptions) : string
        - BuildUppercaseAlphabet(options : PasswordOptions) : string
        - BuildLowercaseAlphabet(options : PasswordOptions) : string
        - RemoveSimilarCharacters(characters : string) : string
        - GetRandomCharacter(characters : string) : char
        - Shuffle(characters : List<char>) : void
    }

    class DictionaryPasswordStrategy <<ConcreteStrategy>> {
        - MaximumAttempts : int
        + Generate(options : PasswordOptions) : string
    }
}

package "Validation" {
    class PasswordOptionsValidator {
        + Validate(options : PasswordOptions) : void
        - ValidateNoSurrogates(value : string, description : string) : void
    }

    class ExternalPasswordValidationService {
        - _verifiers : List<IExternalPasswordVerifier>
        - _logger : ILogger

        + ExternalPasswordValidationService(logger : ILogger)
        + AddVerifier(verifier : IExternalPasswordVerifier) : void
        + IsSafe(password : string) : bool
        - HashPassword(password : string, hashAlgorithmName : string) : string
    }

    interface IExternalPasswordVerifier <<Target Interface>> {
        + Name : string
        + HashAlgorithmName : string
        + IsSafe(passwordHash : string) : bool
    }

    interface IHaveIBeenPwnedApiClient {
        + GetHashSuffixes(hashPrefix : string) : string
    }

    class HaveIBeenPwnedApiClient <<Adaptee>> {
        - _httpClient : HttpClient

        + HaveIBeenPwnedApiClient()
        + GetHashSuffixes(hashPrefix : string) : string
    }

    class HaveIBeenPwnedPasswordVerifierAdapter <<Adapter>> {
        - _apiClient : IHaveIBeenPwnedApiClient

        + Name : string
        + HashAlgorithmName : string
        + HaveIBeenPwnedPasswordVerifierAdapter(apiClient : IHaveIBeenPwnedApiClient)
        + IsSafe(passwordHash : string) : bool
    }
}

' Layout helpers
Program -[hidden]down- PasswordService
PasswordService -[hidden]down- PasswordOptions
PasswordOptions -[hidden]down- IAlphabet
IAlphabet -[hidden]down- CharacterReplacementService
CharacterReplacementService -[hidden]down- IPasswordGenerationStrategy
IPasswordGenerationStrategy -[hidden]down- PasswordOptionsValidator

' Alphabet relationships
LatinAlphabet ..|> IAlphabet
CyrillicAlphabet ..|> IAlphabet
PasswordOptions "1" o-- "0..*" IAlphabet

' Generation relationships
RandomPasswordStrategy ..|> IPasswordGenerationStrategy
DictionaryPasswordStrategy ..|> IPasswordGenerationStrategy

PasswordGenerationFactory ..> IPasswordGenerationStrategy
PasswordGenerationFactory ..> RandomPasswordStrategy
PasswordGenerationFactory ..> DictionaryPasswordStrategy

' Logging relationships
ConsoleLogger ..|> ILogger
ConsoleLogger --> LogLevel

' Validation relationships
ExternalPasswordValidationService --> ILogger
ExternalPasswordValidationService "1" o-- "0..*" IExternalPasswordVerifier

HaveIBeenPwnedApiClient ..|> IHaveIBeenPwnedApiClient
HaveIBeenPwnedPasswordVerifierAdapter ..|> IExternalPasswordVerifier
HaveIBeenPwnedPasswordVerifierAdapter --> IHaveIBeenPwnedApiClient

' Facade relationships
PasswordService --> ILogger
PasswordService *-- PasswordOptionsValidator
PasswordService *-- PasswordGenerationFactory
PasswordService *-- CharacterReplacementService

PasswordService ..> IExternalPasswordVerifier
